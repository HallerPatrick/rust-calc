start:
    stmt

stmt:
    'let' IDENTIFIER '=' expr
    'let' IDENTIFIER '(' identifier-list ')' '=' expr
    expr

expr:
    term

term:
    product '+' term
    product '-' term
    product

product:
    atom '*' product
    atom '/' product
    atom '\' product
    atom '%' product
    atom

atom:
    INTEGER
    FLOAT
    IDENTIFIER
    IDENTIFIER '(' expr-list ')'
    '(' expr ')'
