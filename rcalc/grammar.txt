start:
    stmt

stmt:
    'let' IDENTIFIER '=' expr
    'let' IDENTIFIER '(' identifier-list ')' '=' expr
    'let' IDENTIFIER '(' ')' '=' expr
    expr

identifier-list:
    IDENTIFIER ',' identifier-list
    IDENTIFIER

expr:
    term

term:
    product '+' term
    product '-' term
    product '&' term
    product '|' term
    product '^' term
    product

product:
    molecule '*' product
    molecule '/' product
    molecule '\' product
    molecule '%' product
    molecule '<<' product
    molecule '>>' product
    molecule 

molecule:
    atom 'log' molecule
    atom '**' molecule
    atom

atom:
    INTEGER
    FLOAT
    IDENTIFIER
    IDENTIFIER '(' expr-list ')'
    IDENTIFIER '(' ')'
    'sqrt' atom
    'sin' atom
    'cos' atom
    'tan' atom
    'asin' atom
    'acos' atom
    'atan' atom
    '(' expr ')'

expr-list:
    expr ',' expr-list
    expr
