start:
    stmt

stmt:
    'let' IDENTIFIER '=' expr
    'let' IDENTIFIER '(' identifier-list ')' '=' expr
    'let' IDENTIFIER '(' ')' '=' expr
    expr

identifier-list:
    IDENTIFIER ',' identifier-list
    IDENTIFIER

expr:
    term

term:
    product '+' term
    product '-' term
    product

product:
    atom '*' product
    atom '/' product
    atom '\' product
    atom '%' product
    atom

atom:
    INTEGER
    FLOAT
    IDENTIFIER
    IDENTIFIER '(' expr-list ')'
    IDENTIFIER '(' ')'
    '(' expr ')'

expr-list:
    expr ',' expr-list
    expr
